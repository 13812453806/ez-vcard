373        /**
374         * <p>
375         * Gets the preference value. The lower the number, the more preferred this
376         * property instance is compared with other properties in the same vCard of
377         * the same type. If a property doesn't have a preference value, then it is
378         * considered the least preferred.
379         * </p>
380         * 
381         * <p>
382         * In the vCard below, the address on the second row is the most preferred
383         * because it has the lowest PREF value.
384         * </p>
385         * 
386         * <pre>
387         * ADR;TYPE=work;PREF=2:
388         * ADR;TYPE=work;PREF=1:
389         * ADR;TYPE=home:
390         * </pre>
391         * 
392         * <p>
393         * Preference values must be numeric and must be between 1 and 100.
394         * </p>
395         * 
396         * <p>
397         * <b>Supported versions:</b> {@code 4.0}
398         * </p>
399         * @throws IllegalStateException if the parameter value is malformed and
400         * cannot be parsed
401         * @return the preference value or null if it doesn't exist or null if it
402         * couldn't be parsed into a number
403         */
404        public Integer getPref() {
405                String pref = first(PREF);
406                if (pref == null) {
407                        return null;
408                }
409
410                try {
411                        return Integer.valueOf(pref);
412                } catch (NumberFormatException e) {
413                        throw new IllegalStateException(Messages.INSTANCE.getExceptionMessage(15, PREF), e);
414                }
415        }

541
542                try {

544                        return new double[] { geoUri.getCoordA(), geoUri.getCoordB() };
545                } catch (IllegalArgumentException e) {
546                        throw new IllegalStateException(Messages.INSTANCE.getExceptionMessage(15, GEO), e);

638        /**
639         * <p>
640         * Gets all PID parameter values. PIDs can exist on any property where
641         * multiple instances are allowed (such as EMAIL or ADR, but not N because
642         * only 1 instance of N is allowed).
643         * </p>
644         * <p>
645         * When used in conjunction with the CLIENTPIDMAP property, it allows an
646         * individual property instance to be uniquely identifiable. This feature is
647         * made use of when two different versions of the same vCard have to be
648         * merged together (called "synchronizing").
649         * </p>
650         * <p>
651         * <b>Supported versions:</b> {@code 4.0}
652         * </p>
653         * @throws IllegalStateException if the parameter value is malformed and
654         * cannot be parsed
655         * @return the PID values or empty set if there are none. Index 0 is the
656         * local ID and index 1 is the ID used to reference the CLIENTPIDMAP
657         * property. Index 0 will never be null, but index 1 may be null.
658         */
659        public List<Integer[]> getPids() {
660                List<String> values = get(PID);
661                List<Integer[]> pids = new ArrayList<Integer[]>(values.size());
662                for (String value : values) {
663                        int dot = value.indexOf('.');
664                        String localIdStr, clientPidMapRefStr;
665                        if (dot < 0) {
666                                localIdStr = value;
667                                clientPidMapRefStr = null;
668                        } else {
669                                localIdStr = value.substring(0, dot);
670                                clientPidMapRefStr = (dot < value.length() - 1) ? value.substring(dot + 1) : null;
671                        }
672                        try {
673                                Integer localId = Integer.valueOf(localIdStr);
674                                Integer clientPidMapRef = (clientPidMapRefStr == null) ? null : Integer.valueOf(clientPidMapRefStr);
675                                pids.add(new Integer[] { localId, clientPidMapRef });
676                        } catch (NumberFormatException e) {
677                                throw new IllegalStateException(Messages.INSTANCE.getExceptionMessage(15, PID), e);
678                        }
679                }
680                return pids;
681        }

686         * instances are allowed (such as EMAIL or ADR, but not N because only 1

688         * </p>

691         * individual property instance to be uniquely identifiable. This feature is

852        /**
853         * Checks this parameters list for data consistency problems or deviations
854         * from the spec. These problems will not prevent the vCard from being
855         * written to a data stream, but may prevent it from being parsed correctly
856         * by the consuming application.
857         * @param version the vCard version to validate against
858         * @return a list of warnings or an empty list if no problems were found
859         */
860        public List<Warning> validate(VCardVersion version) {
861                List<Warning> warnings = new ArrayList<Warning>(0);
862
863                /*
864                 * Check for invalid characters in names and values.
865                 */
866                {
867                        BitSet invalidValueChars = new BitSet(128);
868                        invalidValueChars.set(0, 31);
869                        invalidValueChars.set(127);
870                        invalidValueChars.set('\t', false); //allow
871                        invalidValueChars.set('\n', false); //allow
872                        invalidValueChars.set('\r', false); //allow
873                        if (version == VCardVersion.V2_1) {
874                                invalidValueChars.set(',');
875                                invalidValueChars.set('.');
876                                invalidValueChars.set(':');
877                                invalidValueChars.set('=');
878                                invalidValueChars.set('[');
879                                invalidValueChars.set(']');
880                        }
881
882                        CharacterBitSet validNameChars = new CharacterBitSet("-a-zA-Z0-9");
883                        for (Map.Entry<String, List<String>> entry : this) {
884                                String name = entry.getKey();
885
886                                /*
887                                 * Don't check LABEL for 2.1 and 3.0 because this is converted
888                                 * to a property in those version.
889                                 */
890                                if (version != VCardVersion.V4_0 && LABEL.equalsIgnoreCase(name)) {
891                                        continue;
892                                }
893
894                                //check the parameter name
895                                if (!validNameChars.containsOnly(name)) {
896                                        warnings.add(new Warning(26, name));
897                                }
898
899                                //check the parameter value(s)
900                                List<String> values = entry.getValue();
901                                for (String value : values) {
902                                        for (int i = 0; i < value.length(); i++) {
903                                                char c = value.charAt(i);
904                                                if (invalidValueChars.get(c)) {
905                                                        warnings.add(new Warning(25, name, value, (int) c, i));
906                                                        break;
907                                                }
908                                        }
909                                }
910                        }
911                }
912
913                /*
914                 * Check for invalid or unsupported values (e.g. "ENCODING=foo").
915                 */
916                {
917                        int nonStandardCode = 3;
918                        int valueNotSupportedCode = 4;
919
920                        String value = first(CALSCALE);
921                        if (value != null && Calscale.find(value) == null) {
922                                warnings.add(new Warning(nonStandardCode, CALSCALE, value, Calscale.all()));
923                        }
924
925                        value = first(ENCODING);
926                        if (value != null) {
927                                Encoding encoding = Encoding.find(value);
928                                if (encoding == null) {
929                                        warnings.add(new Warning(nonStandardCode, ENCODING, value, Encoding.all()));
930                                } else if (!encoding.isSupportedBy(version)) {
931                                        warnings.add(new Warning(valueNotSupportedCode, ENCODING, value));
932                                }
933                        }
934
935                        value = first(VALUE);
936                        if (value != null) {
937                                VCardDataType dataType = VCardDataType.find(value);
938                                if (dataType == null) {
939                                        warnings.add(new Warning(nonStandardCode, VALUE, value, VCardDataType.all()));
940                                } else if (!dataType.isSupportedBy(version)) {
941                                        warnings.add(new Warning(valueNotSupportedCode, VALUE, value));
942                                }
943                        }
944                }
945
946                /*
947                 * Check for parameters with malformed values.
948                 */
949                {
950                        int malformedCode = 5;
951
952                        try {
953                                getGeo();
954                        } catch (IllegalStateException e) {
955                                warnings.add(new Warning(malformedCode, GEO, first(GEO)));
956                        }
957
958                        try {
959                                getIndex();
960                        } catch (IllegalStateException e) {
961                                warnings.add(new Warning(malformedCode, INDEX, first(INDEX)));
962                        }
963
964                        try {
965                                getPids();
966                        } catch (IllegalStateException e) {
967                                warnings.add(new Warning(malformedCode, PID, first(PID)));
968                        }
969
970                        try {
971                                getPref();
972                        } catch (IllegalStateException e) {
973                                warnings.add(new Warning(malformedCode, PREF, first(PREF)));
974                        }
975                }
976
977                /*
978                 * Check that each parameter is supported by the given vCard version.
979                 */
980                {
981                        int paramNotSupportedCode = 6;
982                        for (Map.Entry<String, Set<VCardVersion>> entry : supportedVersions.entrySet()) {
983                                String name = entry.getKey();
984                                String value = first(name);
985                                if (value == null) {
986                                        continue;
987                                }
988
989                                Set<VCardVersion> versions = entry.getValue();
990                                if (!versions.contains(version)) {
991                                        warnings.add(new Warning(paramNotSupportedCode, name));
992                                }
993                        }
994                }
995
996                /*
997                 * Check that the CHARSET parameter has a character set that is
998                 * supported by this JVM.
999                 */
1000                {
1001                        int invalidCharsetCode = 22;
1002                        String charsetStr = getCharset();
1003                        if (charsetStr != null) {
1004                                try {
1005                                        Charset.forName(charsetStr);
1006                                } catch (IllegalCharsetNameException e) {
1007                                        warnings.add(new Warning(invalidCharsetCode, charsetStr));
1008                                } catch (UnsupportedCharsetException e) {
1009                                        warnings.add(new Warning(invalidCharsetCode, charsetStr));
1010                                }
1011                        }
1012                }
1013
1014                return warnings;
1015        }

1022        @Override
1023        public int hashCode() {
1024                /*
1025                 * Remember: Keys are case-insensitive, key order does not matter, and
1026                 * value order does not matter
1027                 */
1028                final int prime = 31;
1029                int result = 1;
1030
1031                for (Map.Entry<String, List<String>> entry : this) {
1032                        String key = entry.getKey();
1033                        List<String> value = entry.getValue();
1034
1035                        int valueHash = 1;
1036                        for (String v : value) {
1037                                valueHash += v.toLowerCase().hashCode();
1038                        }
1039
1040                        int entryHash = 1;
1041                        entryHash += prime * entryHash + key.toLowerCase().hashCode();
1042                        entryHash += prime * entryHash + valueHash;
1043
1044                        result += entryHash;
1045                }
1046
1047                return result;
1048        }

1050        /**
1051         * <p>
1052         * Determines whether the given object is logically equivalent to this list
1053         * of vCard parameters.
1054         * </p>
1055         * <p>
1056         * vCard parameters are case-insensitive. Also, the order in which they are
1057         * defined does not matter.
1058         * </p>
1059         * @param obj the object to compare to
1060         * @return true if the objects are equal, false if not
1061         */
1062        @Override
1063        public boolean equals(Object obj) {
1064                /*
1065                 * Remember: Keys are case-insensitive, key order does not matter, and
1066                 * value order does not matter
1067                 */
1068                if (this == obj) return true;
1069                if (obj == null) return false;
1070                if (getClass() != obj.getClass()) return false;
1071
1072                VCardParameters other = (VCardParameters) obj;
1073                if (size() != other.size()) return false;
1074
1075                for (Map.Entry<String, List<String>> entry : this) {
1076                        String key = entry.getKey();
1077                        List<String> value = entry.getValue();
1078
1079                        /*
1080                         * The parameters are stored in a ListMultimap. This class never
1081                         * returns null. It returns an empty list when the key is not found.
1082                         * But keep the null check just incase.
1083                         */
1084                        if (value == null) {
1085                                if (other.get(key) != null || !other.containsKey(key)) {
1086                                        return false;
1087                                }
1088                                continue;
1089                        }
1090
1091                        List<String> otherValue = other.get(key);
1092                        if (otherValue == null || value.size() != otherValue.size()) {
1093                                return false;
1094                        }
1095
1096                        List<String> valueLower = new ArrayList<String>(value.size());
1097                        for (String v : value) {
1098                                valueLower.add(v.toLowerCase());
1099                        }
1100                        Collections.sort(valueLower);
1101
1102                        List<String> otherValueLower = new ArrayList<String>(otherValue.size());
1103                        for (String v : otherValue) {
1104                                otherValueLower.add(v.toLowerCase());
1105                        }
1106                        Collections.sort(otherValueLower);
1107
1108                        if (!valueLower.equals(otherValueLower)) {
1109                                return false;
1110                        }
1111                }
1112
1113                return true;
1114        }
