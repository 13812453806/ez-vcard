048/**
049 * Represents a property whose value contains binary data.
050 * @author Michael Angstadt
051 * @param <T> the class used for representing the content type of the resource
052 */
053public abstract class BinaryProperty<T extends MediaTypeParameter> extends VCardProperty implements HasAltId {
054        /**
055         * The decoded data.
056         */
057        protected byte[] data;
058
059        /**
060         * The URL to the resource.
061         */
062        protected String url;
063
064        /**
065         * The content type of the resource (for example, a JPEG image).
066         */
067        protected T contentType;
068
069        public BinaryProperty() {
070                //empty
071        }
072
073        /**
074         * Creates a binary property.
075         * @param url the URL to the resource
076         * @param type the content type
077         */
078        public BinaryProperty(String url, T type) {
079                setUrl(url, type);
080        }
081
082        /**
083         * Creates a binary property.
084         * @param data the binary data
085         * @param type the content type
086         */
087        public BinaryProperty(byte[] data, T type) {
088                setData(data, type);
089        }
090
091        /**
092         * Creates a binary property.
093         * @param in an input stream to the binary data (will be closed)
094         * @param type the content type
095         * @throws IOException if there is a problem reading from the input stream
096         */
097        public BinaryProperty(InputStream in, T type) throws IOException {
098                this(IOUtils.toByteArray(in, true), type);
099        }
100
101        /**
102         * Creates a binary property.
103         * @param file the file containing the binary data
104         * @param type the content type
105         * @throws IOException if there is a problem reading from the file
106         */
107        public BinaryProperty(File file, T type) throws IOException {
108                this(new BufferedInputStream(new FileInputStream(file)), type);
109        }
110
111        /**
112         * Copy constructor.
113         * @param original the property to make a copy of
114         */
115        public BinaryProperty(BinaryProperty<T> original) {
116                super(original);
117                data = (original.data == null) ? null : Arrays.copyOf(original.data, original.data.length);
118                url = original.url;
119                contentType = original.contentType;
120        }
121
122        /**
123         * Gets the binary data of the resource.
124         * @return the binary data or null if there is none
125         */
126        public byte[] getData() {
127                return data;
128        }
129
130        /**
131         * Sets the binary data of the resource.
132         * @param data the binary data
133         * @param type the content type (e.g. "JPEG image")
134         */
135        public void setData(byte[] data, T type) {
136                this.url = null;
137                this.data = data;
138                setContentType(type);
139        }
140
141        /**
142         * Gets the URL to the resource
143         * @return the URL or null if there is none
144         */
145        public String getUrl() {
146                return url;
147        }
148
149        /**
150         * Sets the URL to the resource.
151         * @param url the URL
152         * @param type the content type (e.g. "JPEG image")
153         */
154        public void setUrl(String url, T type) {
155                this.url = url;
156                this.data = null;
157                setContentType(type);
158        }
159
160        /**
161         * Gets the content type of the resource.
162         * @return the content type (e.g. "JPEG image")
163         */
164        public T getContentType() {
165                return contentType;
166        }
167
168        /**
169         * Sets the content type of the resource.
170         * @param contentType the content type (e.g. "JPEG image")
171         */
172        public void setContentType(T contentType) {
173                this.contentType = contentType;
174        }
175
176        /**
177         * Gets the vCard 4.0 TYPE parameter. This should NOT be used to get the
178         * TYPE parameter for 2.1/3.0 vCards. Use {@link #getContentType} instead.
179         * <p>
180         * <b>Supported versions:</b> {@code 4.0}
181         * </p>
182         * @return the TYPE value (typically, this will be either "work" or "home")
183         * or null if it doesn't exist
184         */
185        public String getType() {
186                return parameters.getType();
187        }
188
189        /**
190         * Sets the vCard 4.0 TYPE parameter. This should NOT be used to set the
191         * TYPE parameter for 2.1/3.0 vCards. Use {@link #setContentType} instead.
192         * <p>
193         * <b>Supported versions:</b> {@code 4.0}
194         * </p>
195         * @param type the TYPE value (should be either "work" or "home") or null to
196         * remove
197         */
198        public void setType(String type) {
199                parameters.setType(type);
200        }
201
202        @Override
203        public List<Integer[]> getPids() {
204                return super.getPids();
205        }
206
207        @Override
208        public void addPid(int localId, int clientPidMapRef) {
209                super.addPid(localId, clientPidMapRef);
210        }
211
212        @Override
213        public void removePids() {
214                super.removePids();
215        }
216
217        @Override
218        public Integer getPref() {
219                return super.getPref();
220        }
221
222        @Override
223        public void setPref(Integer pref) {
224                super.setPref(pref);
225        }
226
227        //@Override
228        public String getAltId() {
229                return parameters.getAltId();
230        }
231
232        //@Override
233        public void setAltId(String altId) {
234                parameters.setAltId(altId);
235        }
236
237        @Override
238        protected void _validate(List<Warning> warnings, VCardVersion version, VCard vcard) {
239                if (url == null && data == null) {
240                        warnings.add(new Warning(8));
241                }
242        }
243
244        @Override
245        protected Map<String, Object> toStringValues() {
246                Map<String, Object> values = new LinkedHashMap<String, Object>();
247                values.put("data", (data == null) ? "null" : "length: " + data.length);
248                values.put("url", url);
249                values.put("contentType", contentType);
250                return values;
251        }
252
253        @Override
254        public int hashCode() {
255                final int prime = 31;
256                int result = super.hashCode();
257                result = prime * result + ((contentType == null) ? 0 : contentType.hashCode());
258                result = prime * result + Arrays.hashCode(data);
259                result = prime * result + ((url == null) ? 0 : url.hashCode());
260                return result;
261        }
262
263        @Override
264        public boolean equals(Object obj) {
265                if (this == obj) return true;
266                if (!super.equals(obj)) return false;
267                BinaryProperty<?> other = (BinaryProperty<?>) obj;
268                if (contentType == null) {
269                        if (other.contentType != null) return false;
270                } else if (!contentType.equals(other.contentType)) return false;
271                if (!Arrays.equals(data, other.data)) return false;
272                if (url == null) {
273                        if (other.url != null) return false;
274                } else if (!url.equals(other.url)) return false;
275                return true;
276        }
277}
