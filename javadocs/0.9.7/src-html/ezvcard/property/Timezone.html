042/**
043 * <p>
044 * Defines the timezone that the person lives/works in.
045 * </p>
046 * 
047 * <p>
048 * <b>Code sample</b>
049 * </p>
050 * 
051 * <pre class="brush:java">
052 * VCard vcard = new VCard();
053 * 
054 * Timezone tz = new Timezone(-5, 0, &quot;America/New_York&quot;);
055 * vcard.addTimezone(tz);
056 * 
057 * //using a Java &quot;TimeZone&quot; object
058 * java.util.TimeZone javaTz = java.util.TimeZone.getTimeZone(&quot;America/New_York&quot;);
059 * tz = new Timezone(javaTz);
060 * vcard.addTimezone(tz);
061 * </pre>
062 * 
063 * <p>
064 * <b>Property name:</b> {@code TZ}
065 * </p>
066 * <p>
067 * <b>Supported versions:</b> {@code 2.1, 3.0, 4.0}
068 * </p>
069 * 
070 * @author Michael Angstadt
071 */
072public class Timezone extends VCardProperty implements HasAltId {
073        private UtcOffset offset;
074        private String text;
075
076        /**
077         * Creates a timezone property.
078         * @param text a free-form string representing the timezone, preferably a
079         * timezone ID from the <a
080         * href="http://en.wikipedia.org/wiki/List_of_tz_database_time_zones">Olson
081         * Database</a> (e.g. "America/New_York")
082         */
083        public Timezone(String text) {
084                this(null, text);
085        }
086
087        /**
088         * Creates a timezone property.
089         * @param offset the UTC offset
090         */
091        public Timezone(UtcOffset offset) {
092                this(offset, null);
093        }
094
095        /**
096         * Creates a timezone property.
097         * @param offset the UTC offset
098         * @param text a free-form string representing the timezone, preferably a
099         * timezone ID from the <a
100         * href="http://en.wikipedia.org/wiki/List_of_tz_database_time_zones">Olson
101         * Database</a> (e.g. "America/New_York")
102         */
103        public Timezone(UtcOffset offset, String text) {
104                setOffset(offset);
105                setText(text);
106        }
107
108        /**
109         * Creates a timezone property.
110         * @param timezone the timezone
111         */
112        public Timezone(TimeZone timezone) {
113                this(UtcOffset.parse(timezone), timezone.getID());
114        }
115
116        /**
117         * Gets the UTC offset.
118         * @return the UTC offset or null if not set
119         */
120        public UtcOffset getOffset() {
121                return offset;
122        }
123
124        /**
125         * Sets the UTC offset.
126         * @param offset the UTC offset
127         */
128        public void setOffset(UtcOffset offset) {
129                this.offset = offset;
130        }
131
132        /**
133         * Gets the text portion of the timezone.
134         * @return the free-form string representing the timezone, such as a
135         * timezone ID from the <a
136         * href="http://en.wikipedia.org/wiki/List_of_tz_database_time_zones">Olson
137         * Database</a> (e.g. "America/New_York")
138         */
139        public String getText() {
140                return text;
141        }
142
143        /**
144         * Sets the text portion of the timezone.
145         * @param text a free-form string representing the timezone, preferably a
146         * timezone ID from the <a
147         * href="http://en.wikipedia.org/wiki/List_of_tz_database_time_zones">Olson
148         * Database</a> (e.g. "America/New_York")
149         */
150        public void setText(String text) {
151                this.text = text;
152        }
153
154        /**
155         * Creates a {@link java.util.TimeZone} representation of this class.
156         * @return a {@link TimeZone} object or null if this object contains no
157         * offset data
158         */
159        public TimeZone toTimeZone() {
160                if (text != null) {
161                        TimeZone timezone = VCardDateFormat.parseTimeZoneId(text);
162                        if (timezone != null) {
163                                return timezone;
164                        }
165                }
166
167                if (offset != null) {
168                        String id = (text == null) ? "" : text;
169                        return new SimpleTimeZone((int) offset.getMillis(), id);
170                }
171
172                return null;
173        }
174
175        /**
176         * Gets the TYPE parameter.
177         * <p>
178         * <b>Supported versions:</b> {@code 4.0}
179         * </p>
180         * @return the TYPE value (typically, this will be either "work" or "home")
181         * or null if it doesn't exist
182         */
183        public String getType() {
184                return parameters.getType();
185        }
186
187        /**
188         * Sets the TYPE parameter.
189         * <p>
190         * <b>Supported versions:</b> {@code 4.0}
191         * </p>
192         * @param type the TYPE value (this should be either "work" or "home") or
193         * null to remove
194         */
195        public void setType(String type) {
196                parameters.setType(type);
197        }
198
199        /**
200         * Gets the MEDIATYPE parameter.
201         * <p>
202         * <b>Supported versions:</b> {@code 4.0}
203         * </p>
204         * @return the media type or null if not set
205         */
206        public String getMediaType() {
207                return parameters.getMediaType();
208        }
209
210        /**
211         * Sets the MEDIATYPE parameter.
212         * <p>
213         * <b>Supported versions:</b> {@code 4.0}
214         * </p>
215         * @param mediaType the media type or null to remove
216         */
217        public void setMediaType(String mediaType) {
218                parameters.setMediaType(mediaType);
219        }
220
221        @Override
222        public List<Integer[]> getPids() {
223                return super.getPids();
224        }
225
226        @Override
227        public void addPid(int localId, int clientPidMapRef) {
228                super.addPid(localId, clientPidMapRef);
229        }
230
231        @Override
232        public void removePids() {
233                super.removePids();
234        }
235
236        @Override
237        public Integer getPref() {
238                return super.getPref();
239        }
240
241        @Override
242        public void setPref(Integer pref) {
243                super.setPref(pref);
244        }
245
246        //@Override
247        public String getAltId() {
248                return parameters.getAltId();
249        }
250
251        //@Override
252        public void setAltId(String altId) {
253                parameters.setAltId(altId);
254        }
255
256        @Override
257        protected void _validate(List<Warning> warnings, VCardVersion version, VCard vcard) {
258                if (offset == null && text == null) {
259                        warnings.add(new Warning(8));
260                }
261                if (offset == null && version == VCardVersion.V2_1) {
262                        warnings.add(new Warning(20));
263                }
264        }
265}
