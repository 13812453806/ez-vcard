040/**
041 * Defines the data type of a property's value.
042 * @author Michael Angstadt
043 */
044public class VCardDataType {
045        private static final CaseClasses<VCardDataType, String> enums = new CaseClasses<VCardDataType, String>(VCardDataType.class) {
046                @Override
047                protected VCardDataType create(String value) {
048                        return new VCardDataType(value);
049                }
050
051                @Override
052                protected boolean matches(VCardDataType dataType, String value) {
053                        return dataType.name.equalsIgnoreCase(value);
054                }
055        };
056
057        /**
058         * <p>
059         * A uniform resource locator (e.g. "http://www.example.com/image.jpg").
060         * This data type is only used in 2.1 vCards. All other vCard versions use
061         * {@link #URI}.
062         * </p>
063         * <b>Supported versions:</b> {@code 2.1 (p.18-9)}
064         */
065        public static final VCardDataType URL = new VCardDataType("url", VCardVersion.V2_1);
066
067        /**
068         * <p>
069         * Refers to a MIME entity within an email.
070         * </p>
071         * <b>Supported versions:</b> {@code 2.1 (p.8-9)}
072         */
073        public static final VCardDataType CONTENT_ID = new VCardDataType("content-id", VCardVersion.V2_1);
074
075        /**
076         * <p>
077         * A non-textual value, such as a picture or sound file.
078         * </p>
079         * <b>Supported versions:</b> {@code 3.0}
080         */
081        public static final VCardDataType BINARY = new VCardDataType("binary", VCardVersion.V3_0);
082
083        /**
084         * <p>
085         * A uniform resource identifier (e.g. "http://www.example.com/image.jpg").
086         * 2.1 vCards use {@link #URL} instead.
087         * </p>
088         * <b>Supported versions:</b> {@code 3.0, 4.0}
089         */
090        public static final VCardDataType URI = new VCardDataType("uri", VCardVersion.V3_0, VCardVersion.V4_0);
091
092        /**
093         * <p>
094         * A plain text value.
095         * </p>
096         * <b>Supported versions:</b> {@code 2.1, 3.0, 4.0}
097         */
098        public static final VCardDataType TEXT = new VCardDataType("text");
099
100        /**
101         * <p>
102         * A date that does not have a time component (e.g. "2015-02-16").
103         * </p>
104         * <b>Supported versions:</b> {@code 3.0, 4.0}
105         */
106        public static final VCardDataType DATE = new VCardDataType("date", VCardVersion.V3_0, VCardVersion.V4_0);
107
108        /**
109         * <p>
110         * A time that does not have a date component (e.g. "08:34:00").
111         * </p>
112         * <b>Supported versions:</b> {@code 3.0, 4.0}
113         */
114        public static final VCardDataType TIME = new VCardDataType("time", VCardVersion.V3_0, VCardVersion.V4_0);
115
116        /**
117         * <p>
118         * A date with a time component (e.g. "2015-02-16 08:34:00").
119         * </p>
120         * <b>Supported versions:</b> {@code 3.0, 4.0}
121         */
122        public static final VCardDataType DATE_TIME = new VCardDataType("date-time", VCardVersion.V3_0, VCardVersion.V4_0);
123
124        /**
125         * <p>
126         * Any sort of date/time combination. The value can be a date (e.g.
127         * "2015-02-16"), a time (e.g. "08:34:00"), or a date with a time component
128         * (e.g. "2015-02-16 08:34:00").
129         * </p>
130         * <b>Supported versions:</b> {@code 4.0}
131         */
132        public static final VCardDataType DATE_AND_OR_TIME = new VCardDataType("date-and-or-time", VCardVersion.V4_0);
133
134        /**
135         * <p>
136         * A specific moment in time. Timestamps should be in UTC time.
137         * </p>
138         * <b>Supported versions:</b> {@code 4.0}
139         */
140        public static final VCardDataType TIMESTAMP = new VCardDataType("timestamp", VCardVersion.V4_0);
141
142        /**
143         * <p>
144         * A boolean value ("true" or "false").
145         * </p>
146         * <b>Supported versions:</b> {@code 4.0}
147         */
148        public static final VCardDataType BOOLEAN = new VCardDataType("boolean", VCardVersion.V4_0);
149
150        /**
151         * <p>
152         * An integer value (e.g. "42").
153         * </p>
154         * <b>Supported versions:</b> {@code 4.0}
155         */
156        public static final VCardDataType INTEGER = new VCardDataType("integer", VCardVersion.V4_0);
157
158        /**
159         * <p>
160         * A floating-point value (e.g. "3.14").
161         * </p>
162         * <b>Supported versions:</b> {@code 4.0}
163         */
164        public static final VCardDataType FLOAT = new VCardDataType("float", VCardVersion.V4_0);
165
166        /**
167         * <p>
168         * An offset from UTC time, in hours and minutes (e.g. "-0500").
169         * </p>
170         * <b>Supported versions:</b> {@code 4.0}
171         */
172        public static final VCardDataType UTC_OFFSET = new VCardDataType("utc-offset", VCardVersion.V4_0);
173
174        /**
175         * <p>
176         * A standardized abbreviation for a language (e.g. "en-us" for American
177         * English).
178         * </p>
179         * <b>Supported versions:</b> {@code 4.0}
180         * @see <a href="http://tools.ietf.org/html/rfc5646">RFC 5646</a>
181         */
182        public static final VCardDataType LANGUAGE_TAG = new VCardDataType("language-tag", VCardVersion.V4_0);
183
184        private final String name;
185        private final Set<VCardVersion> supportedVersions;
186
187        /**
188         * @param name the data type name
189         * @param supportedVersions the vCard versions it is supported in
190         */
191        private VCardDataType(String name, VCardVersion... supportedVersions) {
192                this.name = name;
193                if (supportedVersions.length == 0) {
194                        supportedVersions = VCardVersion.values();
195                }
196
197                Set<VCardVersion> set = EnumSet.copyOf(Arrays.asList(supportedVersions));
198                this.supportedVersions = Collections.unmodifiableSet(set);
199        }
200
201        /**
202         * Gets the name of the data type.
203         * @return the name of the data type (e.g. "text")
204         */
205        public String getName() {
206                return name;
207        }
208
209        /**
210         * Determines if the data type is supported by the given vCard version.
211         * @param version the vCard version
212         * @return true if it is supported, false if not
213         */
214        public boolean isSupported(VCardVersion version) {
215                return supportedVersions.contains(version);
216        }
217
218        @Override
219        public String toString() {
220                return name;
221        }
222
223        /**
224         * Searches for a data type that is defined as a static constant in this
225         * class.
226         * @param dataType the data type name (e.g. "text")
227         * @return the data type or null if not found
228         */
229        public static VCardDataType find(String dataType) {
230                return enums.find(dataType);
231        }
232
233        /**
234         * Searches for a data type and creates one if it cannot be found. All
235         * objects are guaranteed to be unique, so they can be compared with
236         * {@code ==} equality.
237         * @param dataType data type name (e.g. "text")
238         * @return the data type
239         */
240        public static VCardDataType get(String dataType) {
241                return enums.get(dataType);
242        }
243
244        /**
245         * Gets all of the data types that are defined as static constants in this
246         * class.
247         * @return the data types
248         */
249        public static Collection<VCardDataType> all() {
250                return enums.all();
251        }
252}
