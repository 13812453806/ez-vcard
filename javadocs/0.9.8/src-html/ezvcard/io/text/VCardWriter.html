057/**
058 * <p>
059 * Writes {@link VCard} objects to a plain-text vCard data stream.
060 * </p>
061 * <p>
062 * <b>Example:</b>
063 * 
064 * <pre class="brush:java">
065 * VCard vcard1 = ...
066 * VCard vcard2 = ...
067 * File file = new File("vcard.vcf");
068 * VCardWriter writer = null;
069 * try {
070 *   writer = new VCardWriter(file, VCardVersion.V3_0);
071 *   writer.write(vcard1);
072 *   writer.write(vcard2);
073 * } finally {
074 *   if (writer != null) writer.close();
075 * }
076 * </pre>
077 * 
078 * </p>
079 * 
080 * <p>
081 * <b>Changing the line folding settings:</b>
082 * 
083 * <pre class="brush:java">
084 * VCardWriter writer = new VCardWriter(...);
085 * 
086 * //disable line folding
087 * writer.getRawWriter().getFoldedLineWriter().setLineLength(null);
088 * 
089 * //change line length
090 * writer.getRawWriter().getFoldedLineWriter().setLineLength(50);
091 * 
092 * //change folded line indent string
093 * writer.getRawWriter().getFoldedLineWriter().setIndent("\t");
094 * 
095 * //change newline character
096 * writer.getRawWriter().getFoldedLineWriter().setNewline("**");
097 * </pre>
098 * 
099 * </p>
100 * @author Michael Angstadt
101 * @see <a href="http://www.imc.org/pdi/vcard-21.rtf">vCard 2.1</a>
102 * @see <a href="http://tools.ietf.org/html/rfc2426">RFC 2426 (3.0)</a>
103 * @see <a href="http://tools.ietf.org/html/rfc6350">RFC 6350 (4.0)</a>
104 */
105public class VCardWriter extends StreamWriter implements Flushable {
106        private final VCardRawWriter writer;
107        private final LinkedList<Boolean> prodIdStack = new LinkedList<Boolean>();
108        private boolean outlookCompatibility = false;
109
110        /**
111         * @param out the output stream to write to
112         * @param targetVersion the version that the vCards should conform to (if
113         * set to "4.0", vCards will be written in UTF-8 encoding)
114         */
115        public VCardWriter(OutputStream out, VCardVersion targetVersion) {
116                this((targetVersion == VCardVersion.V4_0) ? utf8Writer(out) : new OutputStreamWriter(out), targetVersion);
117        }
118
119        /**
120         * @param file the file to write to
121         * @param targetVersion the version that the vCards should conform to (if
122         * set to "4.0", vCards will be written in UTF-8 encoding)
123         * @throws IOException if there's a problem opening the file
124         */
125        public VCardWriter(File file, VCardVersion targetVersion) throws IOException {
126                this(file, false, targetVersion);
127        }
128
129        /**
130         * @param file the file to write to
131         * @param append true to append to the end of the file, false to overwrite
132         * it
133         * @param targetVersion the version that the vCards should conform to (if
134         * set to "4.0", vCards will be written in UTF-8 encoding)
135         * @throws IOException if there's a problem opening the file
136         */
137        public VCardWriter(File file, boolean append, VCardVersion targetVersion) throws IOException {
138                this((targetVersion == VCardVersion.V4_0) ? utf8Writer(file, append) : new FileWriter(file, append), targetVersion);
139        }

145        public VCardWriter(Writer writer, VCardVersion targetVersion) {
146                this.writer = new VCardRawWriter(writer, targetVersion);

174        /**
175         * <p>
176         * Gets whether the writer will apply circumflex accent encoding on
177         * parameter values (disabled by default, only applies to 3.0 and 4.0
178         * vCards). This escaping mechanism allows for newlines and double quotes to
179         * be included in parameter values.
180         * </p>
181         * 
182         * <p>
183         * When disabled, the writer will replace newlines with spaces and double
184         * quotes with single quotes.
185         * </p>
186         * @return true if circumflex accent encoding is enabled, false if not
187         * @see VCardRawWriter#isCaretEncodingEnabled()
188         */
189        public boolean isCaretEncodingEnabled() {

191        }

193        /**
194         * <p>
195         * Sets whether the writer will apply circumflex accent encoding on
196         * parameter values (disabled by default, only applies to 3.0 and 4.0
197         * vCards). This escaping mechanism allows for newlines and double quotes to
198         * be included in parameter values.
199         * </p>
200         * 
201         * <p>
202         * When disabled, the writer will replace newlines with spaces and double
203         * quotes with single quotes.
204         * </p>
205         * @param enable true to use circumflex accent encoding, false not to
206         * @see VCardRawWriter#setCaretEncodingEnabled(boolean)
207         */
208        public void setCaretEncodingEnabled(boolean enable) {
209                writer.setCaretEncodingEnabled(enable);
210        }

212        /**
213         * <p>
214         * Gets whether vCards generated by this writer will be compatible with
215         * Microsoft Outlook mail clients. This setting is disabled by default.
216         * </p>
217         * <p>
218         * Enabling this setting adds an empty line after all base64-encoded
219         * property values for vCard versions 2.1 and 3.0.
220         * </p>
221         * @return true if enabled, false if disabled (defaults to false).
222         */
223        public boolean isOutlookCompatibility() {
224                return outlookCompatibility;
225        }
226
227        /**
228         * <p>
229         * Sets whether vCards generated by this writer should be fully compatible
230         * with Microsoft Outlook mail clients. This setting is disabled by default.
231         * </p>
232         * <p>
233         * Enabling this setting may make the vCards incompatible with other vCard
234         * consumers.
235         * </p>
236         * <p>
237         * Enabling this setting adds an empty line after all base64-encoded
238         * property values for vCards with versions 2.1 and 3.0. This setting has no
239         * effect on 4.0 vCards, or on vCards that do not have any properties with
240         * base64-encoded values.
241         * </p>
242         * @param enable true to enable, false to disable (defaults to false).
243         */
244        public void setOutlookCompatibility(boolean enable) {
245                outlookCompatibility = enable;
246        }

248        @Override
249        @SuppressWarnings({ "rawtypes", "unchecked" })
250        protected void _write(VCard vcard, List<VCardProperty> propertiesToAdd) throws IOException {
251                VCardVersion targetVersion = getTargetVersion();
252                writer.writeBeginComponent("VCARD");
253                writer.writeVersion();
254
255                for (VCardProperty property : propertiesToAdd) {
256                        VCardPropertyScribe scribe = index.getPropertyScribe(property);
257
258                        //marshal the value
259                        String value = null;
260                        VCard nestedVCard = null;
261                        try {
262                                value = scribe.writeText(property, targetVersion);
263                        } catch (SkipMeException e) {
264                                continue;
265                        } catch (EmbeddedVCardException e) {
266                                nestedVCard = e.getVCard();
267                        }
268
269                        //marshal the parameters
270                        VCardParameters parameters = scribe.prepareParameters(property, targetVersion, vcard);
271
272                        //is the value a nested vCard?
273                        if (nestedVCard != null) {
274                                if (targetVersion == VCardVersion.V2_1) {
275                                        //write a nested vCard (2.1 style)
276                                        writer.writeProperty(property.getGroup(), scribe.getPropertyName(), parameters, value);
277                                        prodIdStack.add(addProdId);
278                                        addProdId = false;
279                                        write(nestedVCard);
280                                        addProdId = prodIdStack.removeLast();
281                                } else {
282                                        //write an embedded vCard (3.0 style)
283                                        StringWriter sw = new StringWriter();
284                                        VCardWriter agentWriter = new VCardWriter(sw, targetVersion);
285                                        agentWriter.getRawWriter().getFoldedLineWriter().setLineLength(null);
286                                        agentWriter.getRawWriter().getFoldedLineWriter().setNewline("\n");
287                                        agentWriter.setAddProdId(false);
288                                        agentWriter.setVersionStrict(versionStrict);
289                                        try {
290                                                agentWriter.write(nestedVCard);
291                                        } catch (IOException e) {
292                                                //should never be thrown because we're writing to a string
293                                        } finally {
294                                                IOUtils.closeQuietly(agentWriter);
295                                        }
296
297                                        String vcardStr = sw.toString();
298                                        vcardStr = VCardPropertyScribe.escape(vcardStr);
299                                        writer.writeProperty(property.getGroup(), scribe.getPropertyName(), parameters, vcardStr);
300                                }
301                                continue;
302                        }
303
304                        /*
305                         * Set the property's data type.
306                         * 
307                         * Only add a VALUE parameter if the data type is: (1) not "unknown"
308                         * (2) different from the property's default data type (3) not the
309                         * date/time special case (see code)
310                         */
311                        VCardDataType dataType = scribe.dataType(property, targetVersion);
312                        if (dataType != null) {
313                                VCardDataType defaultDataType = scribe.defaultDataType(targetVersion);
314                                if (dataType != defaultDataType && !isDateTimeValueParameterSpecialCase(defaultDataType, dataType)) {
315                                        parameters.setValue(dataType);
316                                }
317                        }
318
319                        //write the property
320                        writer.writeProperty(property.getGroup(), scribe.getPropertyName(), parameters, value);
321
322                        insertEmptyLineForOutlook(targetVersion, property);
323                }
324
325                writer.writeEndComponent("VCARD");
326        }
327
328        /**
329         * Outlook 2010 requires an empty line after base64 values (at least, some
330         * of the time).
331         * @param targetVersion the vCard version
332         * @param property the property being written
333         * @see <a href="https://github.com/mangstadt/ez-vcard/issues/21">Issue
334         * 21</a>
335         */
336        private void insertEmptyLineForOutlook(VCardVersion targetVersion, VCardProperty property) throws IOException {
337                if (!outlookCompatibility) {
338                        //setting not enabled
339                        return;
340                }
341
342                if (targetVersion == VCardVersion.V4_0) {
343                        //only do this for 2.1 and 3.0 vCards
344                        return;
345                }
346
347                if (!(property instanceof BinaryProperty)) {
348                        //property does not have binary data
349                        return;
350                }
351
352                BinaryProperty<?> binaryProperty = (BinaryProperty<?>) property;
353                if (binaryProperty.getData() == null) {
354                        //property value is not base64-encoded
355                        return;
356                }
357
358                writer.getFoldedLineWriter().writeln("");
359        }
360
361        /**
362         * Determines if the given default data type is "date-and-or-time" and the
363         * given data type is time-based. Properties that meet this criteria should
364         * NOT be given a VALUE parameter.
365         * @param defaultDataType the property's default data type
366         * @param dataType the current property instance's data type
367         * @return true if the default data type is "date-and-or-time" and the data
368         * type is time-based, false otherwise
369         */
370        private boolean isDateTimeValueParameterSpecialCase(VCardDataType defaultDataType, VCardDataType dataType) {
371                //@formatter:off
372                return
373                defaultDataType == VCardDataType.DATE_AND_OR_TIME &&
374                (
375                        dataType == VCardDataType.DATE ||
376                        dataType == VCardDataType.DATE_TIME ||
377                        dataType == VCardDataType.TIME
378                );
379                //@formatter:on
380        }
